% DO NOT change the document class
\documentclass[class=scrartcl]{standalone}
\KOMAoptions{fontsize=11pt}
\usepackage{amsmath}

% All items can be scaled using this variable. This variable HAS to be defined (default 1).
\newcommand{\itemscalefactor}{2}

% Add needed packages and macros here:


% Encoding options
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}

% to create the line for baseline placement so Illustrator does not crop the pdf
\usepackage{tikz}

% dimensions for tikzpicture
\newlength\maxheight
\newlength\diffheight

% configuration for standalone package
\standaloneconfig{border=0pt, multi}

% standalone environment
\newenvironment{lta}{\ignorespaces}{\ignorespacesafterend}
\standaloneenv{lta}

% environment for standard placement
\newcommand{\LaTeXtoAI}[1]{%
    \begin{lta}%
        \scalebox{\itemscalefactor}{#1}%
    \end{lta}%
}

% environment for baseline placement
\newbox\ltabox
\newcommand{\LaTeXtoAIbase}[1]{
	\begin{lta}
	    \scalebox{\itemscalefactor}{%
		\setbox\ltabox\hbox{%
            #1%
        }%
        \begin{tikzpicture}[baseline={(current bounding box.center)}]%
            \pgfmathsetlength{\maxheight}{max(\ht\ltabox,\dp\ltabox)+1pt};
            \pgfmathsetlength{\diffheight}{0.01pt};
            \draw [line width=0.5\diffheight, opacity=0, draw=white](0,\maxheight-\diffheight) --(0,\maxheight);%
            \draw [line width=0.5\diffheight, opacity=0, draw=white](0,-\maxheight+\diffheight) --(0,-\maxheight);%
        \end{tikzpicture}%
        \unhbox\ltabox}%
	\end{lta}%
}




\begin{document}

\begin{lta}%
\begin{tikzpicture}%
\begin{scope}[cm={2,0,1,2,(0,0)}] % Sets the coordinate trafo matrix entries.
\node[inner sep=0pt,transform shape]{\begin{tikzpicture}\node[inner sep=1pt]{ABC};\end{tikzpicture}};
\end{scope}
\end{tikzpicture}%
\end{lta}

\LaTeXtoAI{ABC}

\end{document}
